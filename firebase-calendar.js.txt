// 🔹 Αντιγραφή των στοιχείων από το Firebase project σου

  apiKey: "API_KEY_ΣΟΥ",
  authDomain: "PROJECT_ID.firebaseapp.com",
  projectId: "PROJECT_ID",
  storageBucket: "PROJECT_ID.appspot.com",
  messagingSenderId: "SENDER_ID",
  appId: "APP_ID"
};

// Αρχικοποίηση Firebase
const app = initializeApp(firebaseConfig);
const db = getFirestore(app);

// 🔹 Προσθήκη event στη Firestore
export async function addEvent(eventData) {
  await addDoc(collection(db, "events"), eventData);
  alert("Το event προστέθηκε!");
}

// 🔹 Ανάγνωση όλων των events
export async function loadEvents() {
  const querySnapshot = await getDocs(collection(db, "events"));
  const list = document.getElementById("calendar-events");
  list.innerHTML = "";
  querySnapshot.forEach((doc) => {
    const data = doc.data();
    list.innerHTML += `<li>${data.title} - ${data.date}</li>`;
  });
}
